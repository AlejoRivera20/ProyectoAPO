package org.UI;

import model.GestorMusica;
import javax.swing.*;
import java.awt.*;

public class VentanaPrincipal extends JFrame {
    private GestorMusica gestorMusica;
    private JTextArea areaListas, areaArtistas, areaCanciones;

    public VentanaPrincipal() {
        gestorMusica = new GestorMusica();
        setTitle("Gestor de Colecciones de Música");
        setSize(600, 400);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setLayout(new GridLayout(1, 3));

        areaListas = new JTextArea();
        areaArtistas = new JTextArea();
        areaCanciones = new JTextArea();

        JPanel panelListas = crearPanel("Listas de Reproducción", areaListas);
        JPanel panelArtistas = crearPanel("Artistas", areaArtistas);
        JPanel panelCanciones = crearPanel("Canciones", areaCanciones);

        add(panelListas);
        add(panelArtistas);
        add(panelCanciones);
    }

    private JPanel crearPanel(String titulo, JTextArea area) {
        JPanel panel = new JPanel();
        panel.setLayout(new BorderLayout());
        panel.setBorder(BorderFactory.createTitledBorder(titulo));
        panel.add(new JScrollPane(area), BorderLayout.CENTER);
        return panel;
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                VentanaPrincipal ventana = new VentanaPrincipal();
                ventana.setVisible(true);
            }
        });
    }
}
